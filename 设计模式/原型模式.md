原型模式用于在同一类型的许多对象之间共享属性


```javascript
class Dog {
	constructor(name) {
		this.name = name
	}
}

let dog1 = new Dog("Andy")
let dog2 = new Dog("Bruce")

dog1.__proto__.bark = () => {
	console.log("bark")
}

dog1.bark()     // bark
dog2.bark()     // bark
```

class Dog {
	constructor(name) {
		this.name = name
	}
}

let dog2 = new Dog("Bruce")

Dog.prototype.bark = () => {
	console.log("bark")
}
